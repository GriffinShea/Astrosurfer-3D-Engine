Level:
	DEL:			quit
	ESC:			open menu

	r:			rebuild level
	f:			freeze (depreciated?)
	1:			step one frame while frozen

	p:			switch possession between tank, editor
	o:			switch camera between room, editor

Editor:
	WASD, SPACE, LALT:	x, y, z movement
	LSHIFT, LCTRL:			speed up, down

	MRIGHT:			deselect all
	MLEFT:			select obj by Transf (added to list)
	TAB:			focus obj via iteration of selected objs

	.:			switch between tick, nudge mode
	KP5:			switch between pos, ori, size mode
	KP6, KP4:		manipulate x+, x-
	KP7, KP1:		manipulate y+, y-
	KP8, KP2:		manipulate z+, z-
	KP*, KP/:		double, halve manipulation increment

	KP0:			recalibrate CollTree
				(use after new or moved static obj (COLLFLAG or COLLTERRAIN))

Tank:
	w, s:			move forward, back
	a, d:			turn left, right
	left, right:		strife left, right
